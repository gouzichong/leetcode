# Leetcode 917
    class Solution {
    public:
        string reverseOnlyLetters(string S) {
            char tmp;
            int begin, end;

            if(S.size() < 1)
                return S;

            begin = 0;
            end = S.size() - 1;

            while(begin < end){
                if(!(('a' <= S[end] && S[end] <= 'z') || ('A' <= S[end] && S[end] <= 'Z'))){
                    end--;
                    continue;
                }

                if(!(('a' <= S[begin] && S[begin] <= 'z') || ('A' <= S[begin] && S[begin] <= 'Z'))){
                    begin++;
                    continue;
                }


                tmp = S[end];
                S[end] = S[begin];
                S[begin] = tmp;
                begin++;
                end--;

            }

            return S;

        }
    };
