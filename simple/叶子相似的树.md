# Leetcode 872
    /**
     * Definition for a binary tree node.
     * struct TreeNode {
     *     int val;
     *     TreeNode *left;
     *     TreeNode *right;
     *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
     * };
     */
    class Solution {
    public:
        bool leafSimilar(TreeNode* root1, TreeNode* root2) {
            vector<TreeNode*> leaf1, leaf2;
            stack<TreeNode*> container1, container2;
            TreeNode* tmp; 
            int i;
            container1.push(root1);
            container2.push(root2);

            while(!container1.empty()){
                tmp = container1.top();
                container1.pop();

                if(tmp->right != NULL)
                    container1.push(tmp->right);

                if(tmp->left != NULL)
                    container1.push(tmp->left);

                if(tmp->left == NULL && tmp->right == NULL)
                    leaf1.push_back(tmp);

            }


            while(!container2.empty()){
                tmp = container2.top();
                container2.pop();

                if(tmp->right != NULL)
                    container2.push(tmp->right);

                if(tmp->left != NULL)
                    container2.push(tmp->left);

                if(tmp->left == NULL && tmp->right == NULL)
                    leaf2.push_back(tmp);

            }



            if(leaf1.size() != leaf2.size())
                return false;

            for(i = 0; i < leaf1.size(); i++){
                if(leaf1[i]->val != leaf2[i]->val)
                    return false;
            }


            return true;
        }
    };
