# Leetcode 665
    class Solution {
    public:
        bool checkPossibility(vector<int>& nums) {
            int len = nums.size();
            int flag = 1;
            for(int i = 1; i < len; i++){
                if(nums[i-1] > nums[i]){
                    if(flag == 0){
                        return false;
                    }
                    if(i == 1 || nums[i-2] <= nums[i]){
                        nums[i-1] = nums[i];
                    }else{
                        nums[i] = nums[i-1];
                    }
                    flag--;
                }
            }

            return true;
        }
    };
