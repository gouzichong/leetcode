        //设计图形判断
        int leastInterval(vector<char>& tasks, int n) {
            vector<int> map(26, 0);
            for(char &c : tasks){
                map[c-'A']++;
            }
            sort(map.begin(), map.end());
            int maxJob = map[25] - 1;
            int idle_slots = maxJob * n;
            for(int i = 24; i >= 0 && map[i] > 0; i--){
                idle_slots -= min(map[i],  maxJob);
            }


            return idle_slots > 0 ? idle_slots + tasks.size() : tasks.size();
        }
    };
